document.addEventListener('DOMContentLoaded', () => {
    // 1. Load Menu Data
    if(typeof menuData !== 'undefined'){
        Papa.parse(menuData, {
            header: true,
            skipEmptyLines: true,
            complete: function(results) {
                initApp(results.data);
            },
            error: function(err) {
                console.error("Error parsing CSV:", err);
                document.getElementById('menu-container').innerHTML = '<p class="text-center text-red-500">Failed to load menu.</p>';
            }
        });
    }

    // 2. Navbar Scroll Effect
    const navbar = document.getElementById('navbar');
    let lastScrollTop = 0;
    
    window.addEventListener('scroll', () => {
        let scrollTop = window.pageYOffset || document.documentElement.scrollTop;
        
        if(scrollTop > 300) {
            if (scrollTop > lastScrollTop) {
                navbar.style.transform = 'translateY(-100%)';
            } else {
                navbar.style.transform = 'translateY(0)';
            }
        } else {
            navbar.style.transform = 'translateY(0)';
        }
        lastScrollTop = scrollTop <= 0 ? 0 : scrollTop;
    });

    // 3. Mobile Menu Toggle
    const mobileBtn = document.getElementById('mobile-menu-btn');
    mobileBtn.addEventListener('click', () => {
        document.getElementById('menu-section').scrollIntoView({behavior: 'smooth'});
    });
});

function initApp(items) {
    const grouped = items.reduce((acc, item) => {
        const cat = item.Category ? item.Category.trim() : 'Others';
        if (!acc[cat]) acc[cat] = [];
        acc[cat].push(item);
        return acc;
    }, {});

    renderSidebar(Object.keys(grouped));
    renderMenuGrid(grouped);
    initScrollSpy();
}

function renderSidebar(categories) {
    const navContainer = document.getElementById('category-nav');
    navContainer.innerHTML = '';

    categories.forEach((cat) => {
        const li = document.createElement('li');
        li.className = "flex-shrink-0"; 
        
        const catId = cat.toLowerCase().replace(/\s+/g, '-').replace(/[^\w-]/g, '');

        const btn = document.createElement('button');
        btn.className = `
            w-full text-left px-4 py-3 rounded-xl font-bold text-sm transition-all duration-200
            border border-transparent hover:bg-white hover:shadow-sm text-gray-500 hover:text-brand-black
            whitespace-nowrap lg:whitespace-normal
            category-link
        `;
        btn.dataset.target = catId;
        btn.textContent = cat;

        btn.addEventListener('click', () => {
            const targetSection = document.getElementById(catId);
            if(targetSection) {
                const offset = 100; 
                const bodyRect = document.body.getBoundingClientRect().top;
                const elementRect = targetSection.getBoundingClientRect().top;
                const elementPosition = elementRect - bodyRect;
                const offsetPosition = elementPosition - offset;

                window.scrollTo({
                    top: offsetPosition,
                    behavior: "smooth"
                });
            }
        });

        li.appendChild(btn);
        navContainer.appendChild(li);
    });
}

function renderMenuGrid(groupedItems) {
    const container = document.getElementById('menu-container');
    container.innerHTML = '';

    for (const [category, items] of Object.entries(groupedItems)) {
        const catId = category.toLowerCase().replace(/\s+/g, '-').replace(/[^\w-]/g, '');

        const section = document.createElement('section');
        section.id = catId;
        section.className = "scroll-mt-32 mb-12 category-section";

        const header = document.createElement('div');
        header.className = "flex items-center gap-4 mb-6";
        header.innerHTML = `
            <h2 class="text-3xl font-black text-brand-black uppercase tracking-tight">${category}</h2>
            <div class="h-1 flex-grow bg-gray-200 rounded-full"></div>
        `;

        const grid = document.createElement('div');
        grid.className = "grid grid-cols-1 md:grid-cols-2 gap-6";

        items.forEach(item => {
            const card = document.createElement('div');
            // FIX: Added max-w-full to prevent overflow
            card.className = "bg-white p-4 rounded-2xl shadow-sm border border-gray-100 hover:shadow-md hover:border-brand-yellow/50 transition duration-300 flex gap-4 group max-w-full";
            
            const imgSrc = item.Image && item.Image.trim() !== '' ? item.Image : null;

            let imgHTML = '';
            if(imgSrc) {
                // Image stays fixed size
                imgHTML = `
                    <div class="w-24 h-24 flex-shrink-0 rounded-xl overflow-hidden bg-gray-100">
                        <img src="${imgSrc}" class="w-full h-full object-cover group-hover:scale-110 transition duration-500" alt="${item.Item}">
                    </div>
                `;
            }

            // FIX: Added min-w-0 to the text container
            // This allows the flex child to shrink below its content size, enabling truncation
            card.innerHTML = `
                ${imgHTML}
                <div class="flex flex-col flex-grow justify-between min-w-0">
                    <div>
                        <div class="flex justify-between items-start gap-2">
                            <h4 class="font-bold text-lg text-brand-black leading-tight truncate">${item.Item}</h4>
                            <span class="font-black text-brand-yellow whitespace-nowrap">${item.Price}</span>
                        </div>
                        <p class="text-gray-500 text-sm mt-1 line-clamp-2 break-words">${item.Description}</p>
                    </div>
                    <button class="mt-3 text-xs font-bold text-brand-black uppercase tracking-wide self-start border-b-2 border-brand-yellow hover:bg-brand-yellow hover:border-transparent hover:px-2 hover:rounded-sm transition-all">
                        Add
                    </button>
                </div>
            `;
            grid.appendChild(card);
        });

        section.appendChild(header);
        section.appendChild(grid);
        container.appendChild(section);
    }
}

function initScrollSpy() {
    const sections = document.querySelectorAll('.category-section');
    const navLinks = document.querySelectorAll('.category-link');

    const observerOptions = {
        root: null,
        rootMargin: '-20% 0px -60% 0px',
        threshold: 0
    };

    const observer = new IntersectionObserver((entries) => {
        entries.forEach(entry => {
            if (entry.isIntersecting) {
                navLinks.forEach(link => {
                    link.classList.remove('active-category', 'shadow-md');
                    link.classList.add('text-gray-500');
                });

                const activeId = entry.target.id;
                const activeLink = document.querySelector(`.category-link[data-target="${activeId}"]`);
                
                if (activeLink) {
                    activeLink.classList.remove('text-gray-500', 'hover:bg-white');
                    activeLink.classList.add('active-category', 'shadow-md');
                    activeLink.scrollIntoView({ behavior: 'smooth', block: 'nearest', inline: 'center' });
                }
            }
        });
    }, observerOptions);

    sections.forEach(section => observer.observe(section));
}
